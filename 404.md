<script>
  // Check if we are trying to access a post
  const path = window.location.pathname;
  
  // If URL starts with /posts/ and isn't the index
  if (path.startsWith('/posts/') && path.length > 7) {
    const slug = path.split('/posts/')[1].replace(/\/$/, ''); // Get slug, remove trailing slash
    
    // Redirect to our dynamic viewer
    document.getElementById('error-message').innerText = "Redirecting to post...";
    window.location.href = '/view-post/?slug=' + slug;
  }
</script>
